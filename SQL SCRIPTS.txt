CREATE TABLE ROLE (
	ID INT PRIMARY KEY,
	NAME VARCHAR
);

CREATE TABLE USERS (
	ID INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
	USERNAME VARCHAR,
	EMAIL VARCHAR,
	PASSWORD VARCHAR
)

CREATE TABLE USER_ROLES(
	USER_ID INT REFERENCES USERS(ID),
	ROLE_ID INT REFERENCES ROLE(ID),
	CONSTRAINT USER_ROLES_PK PRIMARY KEY (USER_ID, ROLE_ID)
)
